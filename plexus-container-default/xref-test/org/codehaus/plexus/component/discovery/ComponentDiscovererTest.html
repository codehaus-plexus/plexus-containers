19  import org.codehaus.plexus.PlexusTestCase;
20  import org.codehaus.plexus.classworlds.ClassWorld;
21  import org.codehaus.plexus.classworlds.realm.ClassRealm;
22  import org.codehaus.plexus.component.repository.ComponentDescriptor;
23  import org.codehaus.plexus.component.repository.ComponentSetDescriptor;
24  import org.codehaus.plexus.context.DefaultContext;
25  
26  import java.io.File;
27  import java.util.List;
28  import java.util.Map;
29  import java.util.TreeMap;
30  
31  /**
32   * @author Jason van Zyl
33   *
34   */
35  public class ComponentDiscovererTest
36      extends PlexusTestCase
37  {
38      public void testDefaultComponentDiscoverer()
39          throws Exception
40      {
41          ComponentDiscoverer componentDiscoverer = new DefaultComponentDiscoverer();
42  
43          ClassWorld classWorld = new ClassWorld();
44  
45          ClassRealm core = classWorld.newRealm( "core" );
46  
47          File testClasses = new File( getBasedir(), "target/test-classes" );
48  
49          core.addURL( testClasses.toURL() );
50  
51          File classes = new File( getBasedir(), "target/classes" );
52  
53          core.addURL( classes.toURL() );
54  
55          List<ComponentSetDescriptor> componentSetDescriptors = componentDiscoverer.findComponents( new DefaultContext(), core );
56  
57          ComponentDescriptor<?> componentDescriptor = byImplementation(componentSetDescriptors).get( "org.codehaus.plexus.component.discovery.DefaultDiscoveredComponent" );
58  
59          assertNotNull("componentDescriptor is null", componentDescriptor );
60  
61          assertEquals( "org.codehaus.plexus.component.discovery.DiscoveredComponent", componentDescriptor.getRole() );
62  
63          assertEquals( "org.codehaus.plexus.component.discovery.DefaultDiscoveredComponent", componentDescriptor.getImplementation() );
64      }
65  
66      private static Map<String, ComponentDescriptor<?>> byImplementation(List<ComponentSetDescriptor> descriptorSets) {
67          TreeMap<String, ComponentDescriptor<?>> index = new TreeMap<String, ComponentDescriptor<?>>();
68          for ( ComponentSetDescriptor descriptorSet : descriptorSets )
69          {
70              for ( ComponentDescriptor<?> descriptor : descriptorSet.getComponents() )
71              {
72                  index.put(descriptor.getImplementation(), descriptor);
73              }
74          }
75          return index;
76      }
77  }
