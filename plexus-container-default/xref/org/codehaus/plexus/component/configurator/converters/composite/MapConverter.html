27  import org.codehaus.plexus.component.configurator.ComponentConfigurationException;
28  import org.codehaus.plexus.component.configurator.ConfigurationListener;
29  import org.codehaus.plexus.component.configurator.converters.AbstractConfigurationConverter;
30  import org.codehaus.plexus.component.configurator.converters.lookup.ConverterLookup;
31  import org.codehaus.plexus.component.configurator.expression.ExpressionEvaluator;
32  import org.codehaus.plexus.configuration.PlexusConfiguration;
33  
34  import java.util.Map;
35  import java.util.Properties;
36  import java.util.TreeMap;
37  
38  /**
39   * Converter for <code>java.util.Properties</code>.
40   *
41   * @author <a href="mailto:michal@codehaus.org">Michal Maczka</a>
42   */
43  public class MapConverter
44      extends AbstractConfigurationConverter
45  {
46      public boolean canConvert( Class type )
47      {
48          return Map.class.isAssignableFrom( type ) && !Properties.class.isAssignableFrom( type );
49      }
50  
51      public Object fromConfiguration( ConverterLookup converterLookup, PlexusConfiguration configuration, Class type,
52                                       Class baseType, ClassLoader classLoader, ExpressionEvaluator expressionEvaluator,
53                                       ConfigurationListener listener )
54          throws ComponentConfigurationException
55      {
56          Object retValue;
57  
58          String expression = configuration.getValue( null );
59  
60          if ( expression == null )
61          {
62              expression = configuration.getAttribute( "default-value", null );
63          }
64  
65          if ( expression == null )
66          {
67              Map map = new TreeMap();
68  
69              PlexusConfiguration[] children = configuration.getChildren();
70  
71              for (PlexusConfiguration child : children) {
72                  String name = child.getName();
73  
74                  map.put(name, fromExpression(child, expressionEvaluator));
75              }
76              retValue = map;
77          }
78          else
79          {
80              retValue = fromExpression( configuration, expressionEvaluator );
81          }
82          return retValue;
83      }
84  
85  }
