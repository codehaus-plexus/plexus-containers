27  import java.text.DateFormat;
28  import java.text.ParseException;
29  import java.text.SimpleDateFormat;
30  import java.util.Date;
31  
32  public class DateConverter extends AbstractBasicConverter {
33      /***
34       * @todo DateFormat is not thread safe!
35       */
36      private static final DateFormat[] formats = {
37          new SimpleDateFormat("yyyy-MM-dd HH:mm:ss.S a"), new SimpleDateFormat("yyyy-MM-dd HH:mm:ssa")
38      };
39  
40      public boolean canConvert(Class type) {
41          return type.equals(Date.class);
42      }
43  
44      public Object fromString(String str) {
45          for (DateFormat format : formats) {
46              try {
47                  return format.parse(str);
48              } catch (ParseException e) {
49                  // no worries, let's try the next format.
50              }
51          }
52  
53          return null;
54      }
55  
56      public String toString(Object obj) {
57          Date date = (Date) obj;
58          return formats[0].format(date);
59      }
60  }
