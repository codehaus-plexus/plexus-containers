27  import org.codehaus.plexus.component.configurator.ComponentConfigurationException;
28  import org.codehaus.plexus.component.configurator.ConfigurationListener;
29  import org.codehaus.plexus.component.configurator.converters.AbstractConfigurationConverter;
30  import org.codehaus.plexus.component.configurator.converters.lookup.ConverterLookup;
31  import org.codehaus.plexus.component.configurator.expression.ExpressionEvaluator;
32  import org.codehaus.plexus.configuration.PlexusConfiguration;
33  
34  /**
35   * Converter for  <code>org.codehaus.plexus.configuration.PlexusConfiguration</code>
36   *
37   * @author <a href="mailto:michal@codehaus.org">Michal Maczka</a>
38   */
39  public class PlexusConfigurationConverter extends AbstractConfigurationConverter {
40      public boolean canConvert(Class type) {
41          return PlexusConfiguration.class.isAssignableFrom(type);
42      }
43  
44      public Object fromConfiguration(
45              ConverterLookup converterLookup,
46              PlexusConfiguration configuration,
47              Class type,
48              Class baseType,
49              ClassLoader classLoader,
50              ExpressionEvaluator expressionEvaluator,
51              ConfigurationListener listener)
52              throws ComponentConfigurationException {
53          return configuration;
54      }
55  }
