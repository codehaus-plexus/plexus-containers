19  import org.codehaus.plexus.PlexusConstants;
20  import org.codehaus.plexus.component.factory.ComponentFactoryManager;
21  import org.codehaus.plexus.context.ContextException;
22  import org.codehaus.plexus.personality.plexus.lifecycle.phase.Contextualizable;
23  
24  /**
25   * @author Jason van Zyl
26   */
27  public class InitializeComponentFactoryManagerPhase
28      extends AbstractCoreComponentInitializationPhase
29  {
30      public void initializeCoreComponent( ContainerInitializationContext context )
31          throws ContainerInitializationException
32      {
33          ComponentFactoryManager componentFactoryManager = context.getContainerConfiguration().getComponentFactoryManager();
34  
35          if ( componentFactoryManager instanceof Contextualizable )
36          {
37              //TODO: this is wrong here jvz.
38              context.getContainer().getContext().put( PlexusConstants.PLEXUS_KEY, context.getContainer() );
39  
40              try
41              {
42                  ( (Contextualizable) componentFactoryManager).contextualize( context.getContainer().getContext() );
43              }
44              catch ( ContextException e )
45              {
46                  throw new ContainerInitializationException( "Error contextualization component factory manager.", e );
47              }
48          }
49  
50          context.getContainer().setComponentFactoryManager( componentFactoryManager );
51      }
52  }
