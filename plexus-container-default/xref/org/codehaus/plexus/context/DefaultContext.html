19  import java.util.Collections;
20  import java.util.Map;
21  import java.util.Map.Entry;
22  import java.util.concurrent.ConcurrentHashMap;
23  import java.util.concurrent.ConcurrentMap;
24  import java.util.concurrent.atomic.AtomicBoolean;
25  
26  /**
27   * Default implementation of Context.
28   *
29   * This implementation is a static hierarchial store. It has the normal <code>get()</code>
30   * and <code>put</code> methods. The <code>hide</code> method will hide a property. When
31   * a property has been hidden the containerContext will not search in the parent containerContext for the value.
32   *
33   * @author <a href="mailto:dev@avalon.codehaus.org">Avalon Development Team</a>
34   */
35  public class DefaultContext
36      implements Context
37  {
38      /**
39       * Context data.
40       */
41      private final ConcurrentMap<Object, Object> contextData = new ConcurrentHashMap<Object, Object>();
42  
43      /**
44       * Is the containerContext read only.
45       */
46      private final AtomicBoolean readOnly = new AtomicBoolean(false);
47  
48      /**
49       * Create an empty Context.
50       */
51      public DefaultContext()
52      {
53      }
54  
55      /**
56       * Create a Context with specified data.  The specified data is copied into the context so any subsequent updates
57       * to supplied map are not reflected in this context.  Additionally, changes to this context are not reflected in
58       * the specified map.
59       *
60       * @param contextData the containerContext data
61       */
62      public DefaultContext( Map<Object, Object> contextData )
63      {
64          if ( contextData == null )
65          {
66              throw new NullPointerException( "contextData is null" );
67          }
68  
69          // check for nulls in key and value
70          for ( Entry<Object, Object> entry : contextData.entrySet() )
71          {
72              Object key = entry.getKey();
73              Object value = entry.getValue();
74              if ( key == null )
75              {
76                  throw new IllegalArgumentException( "Key is null" );
77              }
78              if ( value != null )
79              {
80                  this.contextData.put( key, value );
81              }
82          }
83      }
84  
85      public boolean contains( Object key )
86      {
87          Object data = contextData.get( key );
88  
89          return data != null;
90      }
91  
92      public Object get( Object key )
93          throws ContextException
94      {
95          Object data = contextData.get( key );
96  
97          if ( data == null )
98          {
99              // There is no data for the key
100             throw new ContextException( "Unable to resolve context key: " + key );
101         }
102         return data;
103     }
104 
105     public void put( Object key, Object value )
106         throws IllegalStateException
107     {
108         checkWriteable();
109 
110         // check for a null key
111         if (key == null)
112         {
113             throw new IllegalArgumentException("Key is null");
114         }
115 
116         if ( value == null )
117         {
118             contextData.remove( key );
119         }
120         else
121         {
122             contextData.put( key, value );
123         }
124     }
125 
126     public void hide( Object key )
127         throws IllegalStateException
128     {
129         checkWriteable();
130         
131         contextData.remove( key );
132     }
133 
134     /**
135      * Utility method to retrieve containerContext data
136      *
137      * @return the containerContext data
138      */
139     public Map<Object,Object> getContextData()
140     {
141         return Collections.unmodifiableMap( contextData );
142 
143     }
144 
145     /**
146      * Make the containerContext read-only.
147      * Any attempt to write to the containerContext via put()
148      * will result in an IllegalStateException.
149      */
150     public void makeReadOnly()
151     {
152         readOnly.set( true );
153     }
154 
155     /**
156      * Utility method to check if containerContext is writeable and if not throw exception.
157      *
158      * @throws java.lang.IllegalStateException if containerContext is read only
159      */
160     protected void checkWriteable()
161         throws IllegalStateException
162     {
163         if ( readOnly.get() )
164         {
165             throw new IllegalStateException( "Context is read only and can not be modified" );
166         }
167     }
168     
169     public String toString()
170     {
171         return contextData.toString();
172     }
173 }
