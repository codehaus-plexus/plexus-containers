27  import org.codehaus.plexus.metadata.merge.MergeException;
28  import org.jdom.Element;
29  
30  /**
31   * @author <a href='mailto:rahul.thakur.xdev@gmail.com'>Rahul Thakur</a>
32   * @version $Id$
33   */
34  public class RequirementElement
35      extends AbstractMergeableElement
36  {
37      static final DescriptorTag TAG = new DescriptorTag( "requirement", true, RequirementElement.class );
38  
39      public RequirementElement( Element element )
40      {
41          super( element );
42      }
43  
44      public DescriptorTag[] getAllowedTags()
45      {
46          return new DescriptorTag[] { ComponentElement.ROLE, ComponentElement.ROLE_HINT, ComponentElement.FIELD_NAME,
47              ComponentElement.OPTIONAL };
48      }
49  
50      public void merge( Mergeable me )
51          throws MergeException
52      {
53          super.merge( me );
54      }
55  
56      protected boolean isExpectedElementType( Mergeable me )
57      {
58          return me instanceof RequirementElement;
59      }
60  }
