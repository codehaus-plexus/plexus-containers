27  import java.util.ArrayList;
28  import java.util.List;
29  
30  import org.jdom2.Element;
31  
32  /**
33   * @author <a href='mailto:rahul.thakur.xdev@gmail.com'>Rahul Thakur</a>
34   */
35  public class RequirementsElement extends AbstractMergeableElementList {
36      public static final DescriptorTag TAG = new DescriptorTag("requirements", true, RequirementsElement.class);
37  
38      public RequirementsElement(Element element) {
39          super(element);
40      }
41  
42      protected boolean isExpectedElementType(Mergeable me) {
43          return me instanceof RequirementsElement;
44      }
45  
46      public DescriptorTag[] getAllowedTags() {
47          return new DescriptorTag[] {RequirementElement.TAG};
48      }
49  
50      protected List getElementNamesForConflictResolution(List defaultList) {
51          // we return the keys that we know we want to lookup to identify and
52          // resolve conflicts.
53          List l = new ArrayList();
54          l.add(ComponentElement.ROLE.getTagName());
55          // TODO: add this back, but a test will fail (based on a role with no hint, which shouldn't be legal)
56          //        l.add( ComponentElement.ROLE_HINT.getTagName() );
57          return l;
58      }
59  
60      protected String getTagNameForRecurringMergeable() {
61          return RequirementElement.TAG.getTagName();
62      }
63  }
